@name a missile that explodes mid air
@inputs Launch
@outputs 
@persist A
@trigger 

runOnTick(1)
if(Launch){A=1
    timer("boomidair",2000)
}
if(clk("boomidair")){entity():isWeldedTo():propBreak()}

if(A){entity():isWeldedTo():applyForce(((entity():isWeldedTo():pos()+vec(0,0,100))-entity():isWeldedTo():pos())*30)}
